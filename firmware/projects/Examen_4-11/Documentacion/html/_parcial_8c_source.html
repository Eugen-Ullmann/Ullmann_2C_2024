82static void medicion_vehiculos(void *pvParameters)
83{
84 while (1)
85 {
86 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
87 distancia_vehiculos = HcSr04ReadDistanceInCentimeters();
88
89 distancia_vehiculos = distancia_vehiculos * 100; //Convierto los valores a metros
90
91 if (distancia_vehiculos > 5)
92 LedOn(LED_1);
93
94 else
95
96 LedsOffAll();
97
98 if (distancia_vehiculos < 5 && distancia_vehiculos > 3)
99 {
100 LedOn(LED_1);
101 LedOn(LED_2);
102 UartSendString(UART_CONNECTOR, "Precaución, vehículo cerca ");
103 UartSendString(UART_CONNECTOR, "\r\n");
104 while (distancia_vehiculos < 5 && distancia_vehiculos > 3)
105 {
106 GPIOOn(Buzzer);
107 vTaskDelay(pdMS_TO_TICKS(1000)); // espera 1 segundo
108 GPIOOff(Buzzer);
109 }
110 }
111 else
112 LedsOffAll();
113 if (distancia_vehiculos < 3)
114 {
115 LedOn(LED_1);
116 LedOn(LED_2);
117 LedOn(LED_3);
118 UartSendString(UART_CONNECTOR, "Peligro, vehículo cerca");
119 UartSendString(UART_CONNECTOR, "\r\n");
120
121 while (distancia_vehiculos < 3)
122 {
123 GPIOOn(Buzzer);
124 vTaskDelay(pdMS_TO_TICKS(500)); // espera 0.5 segundos
125 GPIOOff(Buzzer);
126 }
127 }

129 LedsOffAll();
130 }

137static void detector_caidas(void *pvParameters)
138{
139 while (1)
140 {
141 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
142 AnalogInputReadSingle(CH0, &lectura_eje_X);

152 }
